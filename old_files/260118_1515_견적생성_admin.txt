  <script>
    const fmt = (n) => (Number(n || 0)).toLocaleString() + "원";
    const clampNum = (v) => isFinite(v) ? v : 0;

    function floorTo100(n) {
      n = clampNum(Number(n || 0));
      return Math.floor(n / 100) * 100;
    }

    function parseMoneyText(v) {
      if (v === null || v === undefined) return 0;
      return clampNum(Number(String(v).replace(/[^0-9.-]/g, ""))) || 0;
    }

    function formatMoneyInput(el) {
      const num = parseMoneyText(el.value);
      el.value = num ? num.toLocaleString() : (String(el.value).trim() === "" ? "" : "0");
    }

    function calculatePMT(P, rate, months) {
      if (P <= 0) return 0;
      const r = rate / 100 / 12;
      const n = months;
      const pmt = P * r * Math.pow(1 + r, n) / (Math.pow(1 + r, n) - 1);
      return floorTo100(pmt);
    }

    let g_finalEstimate = 0;
    let g_material = 0;

    function recalcBenefitAndMargin() {
      const discountRate = clampNum(Number(document.getElementById("discountRate").value || 0));
      const supplyCost = parseMoneyText(document.getElementById("supplyCost").value);
      const extraDiscount = parseMoneyText(document.getElementById("extraDiscount").value);

      const discountAmt = g_finalEstimate * (discountRate / 100);
      const finalBenefit = Math.max(0, Math.round(g_finalEstimate - discountAmt - extraDiscount));

      const marginAmt = Math.round(finalBenefit - supplyCost);
      const marginRate = finalBenefit > 0 ? (marginAmt / finalBenefit) : 0;

      // UI
      document.getElementById("v_final_benefit").innerText = fmt(finalBenefit);
      document.getElementById("v_discount_breakdown").innerText =
        `할인(${discountRate}%): -${fmt(Math.round(discountAmt))} / 추가할인: -${fmt(extraDiscount)}`;

      document.getElementById("v_margin_amt").innerText = fmt(marginAmt);
      document.getElementById("v_margin_rate").innerText = (marginRate * 100).toFixed(2) + "%";

      // 구독료 (혜택가 기준)
      const ir = 10;
      const s24 = calculatePMT(finalBenefit, ir, 24);
      const s36 = calculatePMT(finalBenefit, ir, 36);
      const s48 = calculatePMT(finalBenefit, ir, 48);
      const s60 = calculatePMT(finalBenefit, ir, 60);
      document.getElementById("v_sub_24").innerText = fmt(s24) + "/월";
      document.getElementById("v_sub_36").innerText = fmt(s36) + "/월";
      document.getElementById("v_sub_48").innerText = fmt(s48) + "/월";
      document.getElementById("v_sub_60").innerText = fmt(s60) + "/월";

      // hidden
      document.getElementById("h_final_benefit").value = finalBenefit;
      document.getElementById("h_margin_amt").value = marginAmt;
      document.getElementById("h_margin_rate").value = (marginRate * 100).toFixed(4); // 시트엔 %로 저장

      document.getElementById("h_sub_24").value = s24;
      document.getElementById("h_sub_36").value = s36;
      document.getElementById("h_sub_48").value = s48;
      document.getElementById("h_sub_60").value = s60;
    }

    // 엑셀 분석/계산만 수행하고 화면/hidden 반영 (DB 저장은 하지 않음)
    
    // 엑셀 분석/계산만 수행하고 화면/hidden에 반영 (엑셀 선택 즉시 자동 실행)
    function analyzeExcelAndFill(after) {
      const status = document.getElementById("status");
      const excelInputEl = document.getElementById("excelFile");
      if (!excelInputEl || !excelInputEl.files || !excelInputEl.files[0]) {
        if (status) status.innerText = "엑셀 파일을 선택해주세요.";
        excelReady = false;
        refreshButtons();
        if (typeof after === "function") after(false);
        return;
      }

      const file = excelInputEl.files[0];
      if (status) status.innerText = "엑셀 분석 중...";

      const toStr = (v) => (v === null || v === undefined) ? "" : String(v).trim();

      // 셀 값(문자/숫자/콤마 포함)을 숫자로 변환
      const parseCellMoney = (v) => {
        if (typeof v === "number") return isFinite(v) ? v : 0;
        const s = toStr(v).replace(/[^0-9.\-]/g, "");
        const n = Number(s);
        return isFinite(n) ? n : 0;
      };

      const reader = new FileReader();

      reader.onerror = () => {
        excelReady = false;
        refreshButtons();
        if (status) status.innerText = "엑셀 파일을 읽을 수 없습니다. 다른 파일로 다시 시도해주세요.";
        if (typeof after === "function") after(false);
      };

      reader.onload = (evt) => {
        try {
          const data = new Uint8Array(evt.target.result);
          const wb = XLSX.read(data, { type: "array" });
          const sheetName = wb.SheetNames && wb.SheetNames.length ? wb.SheetNames[0] : null;
          if (!sheetName) throw new Error("시트를 찾을 수 없습니다.");

          const ws = wb.Sheets[sheetName];
          const rows = XLSX.utils.sheet_to_json(ws, { header: 1, defval: "" }); // 2D array
          if (!rows || rows.length < 2) throw new Error("엑셀 데이터가 비어있습니다.");

          // 헤더 후보: 첫 20행 중에서 '설치'/'위치' 또는 금액 관련 키워드가 포함된 행을 헤더로 선택
          const headerKeywords = /(설치|위치|금액|공급|단가|합계|총액|판매)/;
          let headerRowIdx = 0;
          for (let i = 0; i < Math.min(20, rows.length); i++) {
            const joined = rows[i].map(toStr).join(" ");
            if (headerKeywords.test(joined)) { headerRowIdx = i; break; }
          }

          const header = rows[headerRowIdx].map(toStr);
          const body = rows.slice(headerRowIdx + 1);

          // 설치위치 열 인덱스 (요청: 기본 C열 = 2)
          const findIdx = (regexList) => {
            for (let i = 0; i < header.length; i++) {
              const h = header[i];
              for (const rx of regexList) {
                if (rx.test(h)) return i;
              }
            }
            return -1;
          };

          let locIdx = findIdx([/설치\s*위치/, /설치\s*장소/, /설치.*위치/, /위치/]);
          if (locIdx < 0) locIdx = 2; // 기본 C열

          // 금액 열 인덱스 (헤더 우선 → 없으면 숫자 밀도 최대 열)
          let amtIdx = findIdx([/금액/, /공급/, /단가/, /합계/, /총액/, /판매/]);
          if (amtIdx < 0) {
            // 숫자 값이 가장 많이 들어있는 열을 금액 열로 추정
            const colCounts = [];
            const maxCols = Math.max(...body.map(r => r.length), header.length);
            for (let c = 0; c < maxCols; c++) {
              let cnt = 0;
              for (let r = 0; r < Math.min(body.length, 2000); r++) {
                const n = parseCellMoney(body[r][c]);
                if (n !== 0) cnt++;
              }
              colCounts[c] = cnt;
            }
            amtIdx = colCounts.indexOf(Math.max(...colCounts));
          }

          // 상세내역(JSON): 헤더 기반으로 객체화 + 금액/설치위치 확실히 포함
          const items = [];
          let sumMaterial = 0;
          let sumEtc = 0;

          for (let r = 0; r < body.length; r++) {
            const row = body[r];
            // 완전 빈 행 스킵
            if (!row || row.every(v => toStr(v) === "")) continue;

            const loc = toStr(row[locIdx]);
            const amt = parseCellMoney(row[amtIdx]);

            // 금액이 0이면 상세는 남겨도 되지만 합계에는 영향 없음
            const isEtc = /기타/.test(loc);

            if (amt > 0) {
              if (isEtc) sumEtc += amt;
              else sumMaterial += amt;
            }

            const obj = {};
            for (let c = 0; c < header.length; c++) {
              const key = header[c] || `COL_${c+1}`;
              obj[key] = row[c];
            }
            obj["설치위치"] = loc;
            obj["금액"] = amt;
            obj["_구분"] = isEtc ? "기타" : "자재";
            items.push(obj);
          }

          // 부가세 포함(×1.1) + 100원 단위 절삭
          const g_material = floorTo100(sumMaterial * 1.1);
          const g_etc = floorTo100(sumEtc * 1.1);
          const g_finalEstimate = g_material + g_etc;

          // 화면 표시
          document.getElementById("v_material").innerText = fmt(g_material);
          document.getElementById("v_etc").innerText = fmt(g_etc);
          document.getElementById("v_final_est").innerText = fmt(g_finalEstimate);

          // hidden 값 저장 (원 단위 숫자)
          document.getElementById("h_material").value = String(g_material);
          document.getElementById("h_etc").value = String(g_etc);
          document.getElementById("h_final_est").value = String(g_finalEstimate);
          document.getElementById("h_items").value = JSON.stringify(items);

          // 엑셀 준비 완료 플래그
          excelReady = true;

          // 혜택/마진/구독료 갱신
          recalcBenefitAndMargin();

          // 버튼 상태 갱신
          refreshButtons();

          if (status) status.innerText = "엑셀 분석 완료! 마진 설정 후 PDF 등록 → [등록] 진행하세요.";
          if (typeof after === "function") after(true);
        } catch (err) {
          excelReady = false;
          refreshButtons();
          if (status) status.innerText = "엑셀 분석 실패: " + (err && err.message ? err.message : String(err));
          if (typeof after === "function") after(false);
        }
      };

      // ✅ 실제 읽기 실행 (이게 없으면 자동 계산이 절대 안 됨)
      reader.readAsArrayBuffer(file);
    }


    function handleFormSubmit(e) {
      e.preventDefault();
      if (!excelInputEl || !excelInputEl.files || !excelInputEl.files[0]) {
        alert("엑셀 파일을 먼저 선택해주세요.");
        return;
      }
      if (!pdfInputEl || !pdfInputEl.files || !pdfInputEl.files[0]) {
        alert("PDF 파일을 선택해주세요.");
        return;
      }

      // 엑셀이 아직 분석되지 않았다면 먼저 분석 후 저장
      if (!excelReady) {
        analyzeExcelAndFill(() => doSaveToDB());
        return;
      }

      // 마진/혜택/구독료는 입력 변경 시 자동 반영되므로 그대로 저장
      doSaveToDB();
    }
