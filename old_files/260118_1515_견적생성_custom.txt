    const fmt = (n) => Number(n).toLocaleString() + "원";
    
    // 글로벌 데이터 저장소
    let g_appliances = []; 
    let g_banners = [];

    function formatPhone(phone) {
      if (!phone) return "";
      var clean = phone.replace(/[^0-9]/g, '');
      return clean.replace(/^(\d{2,3})(\d{3,4})(\d{4})$/, '$1-$2-$3');
    }

    function calculatePMT(P, rate, months) {
      if(P <= 0) return 0;
      const r = rate / 100 / 12; 
      const n = months;
      const pmt = P * r * Math.pow(1+r, n) / (Math.pow(1+r, n) - 1);
      return Math.floor(pmt / 100) * 100;
    }

    function doSearch() {
      const name = document.getElementById('inputName').value;
      const phone = document.getElementById('inputPhone').value;
      const msg = document.getElementById('msg');
      if(!name || !phone) { msg.innerText = "정보를 모두 입력해주세요."; return; }
      msg.innerText = "조회 중입니다...";
      
      google.script.run.withSuccessHandler(res => {
        if(res.success) { 
          // Config 데이터 전역 변수에 저장
          g_appliances = res.config.appliances || [];
          g_banners = res.config.banners || [];
          renderData(res.data, phone); 
        } 
        else { msg.innerText = res.message; }
      }).searchQuote(name, phone);
    }

    function renderData(data, inputPhone) {

      } else {
        statusEl.className = "mr-2 text-[10px] md:text-xs font-bold text-gray-500 bg-gray-100 px-2 py-1 rounded border border-gray-300";
      }

      document.getElementById('resName').innerText = data.name;
      document.getElementById('resPhone').innerText = formatPhone(inputPhone); 
      document.getElementById('resAddress').innerText = data.address;
      
      const btnPdf = document.getElementById('btnPdf');
      if(data.pdfUrl) {

        btnPdf.classList.add('opacity-50', 'cursor-not-allowed');
      }

      // 1. 가격 표시 (일시불) - DB에 저장된 최종견적가/최종혜택가 기준
const originalPrice = Number(data.finalEstimate || data.originalPrice || 0);
const discountRate = Number(data.discountRate || 0);
const extraDiscount = Number(data.extraDiscount || 0);
const finalBenefit = Number(data.finalBenefit || 0);

const discountAmount = Math.max(0, originalPrice - finalBenefit);

document.getElementById('opt1_original').innerText = fmt(originalPrice);
document.getElementById('opt1_rate').innerText = discountRate;
document.getElementById('opt1_discount').innerText = fmt(discountAmount);
document.getElementById('opt1_final').innerText = fmt(finalBenefit);


      // 2. 구독료 표시 (DB 저장값 우선)
if (data.sub24 || data.sub36 || data.sub48 || data.sub60) {
  document.getElementById('sub_24').innerText = fmt(Number(data.sub24 || 0)) + "/월";
  document.getElementById('sub_36').innerText = fmt(Number(data.sub36 || 0)) + "/월";
  document.getElementById('sub_48').innerText = fmt(Number(data.sub48 || 0)) + "/월";
  document.getElementById('sub_60').innerText = fmt(Number(data.sub60 || 0)) + "/월";
} else {
  // fallback (구버전 데이터 호환)
  const ir = 10;
  const base = Number(data.finalBenefit || originalPrice || 0);
  document.getElementById('sub_24').innerText = fmt(calculatePMT(base, ir, 24)) + "/월";
  document.getElementById('sub_36').innerText = fmt(calculatePMT(base, ir, 36)) + "/월";
  document.getElementById('sub_48').innerText = fmt(calculatePMT(base, ir, 48)) + "/월";
  document.getElementById('sub_60').innerText = fmt(calculatePMT(base, ir, 60)) + "/월";
}


      // 3. 상세 품목 렌더링
      renderItems(data.items);
      document.getElementById('detailTotal').innerText = fmt(originalPrice);

      // 4. 배너 렌더링
      renderBanners();
    }

    function renderItems(items) {
      const container = document.getElementById('itemContainer');
      container.innerHTML = "";

      items.forEach((item, idx) => {
        if(item.loc.trim() === '비고') return;
        let isEtc = (item.loc === '기타');
        let displaySpecLabel = isEtc ? '구분' : '모델명';
        let displaySpecValue = item.spec || '-'; 
        
        let thickInfo = '';
        let glassInfo = '';

        if (!isEtc) {
          thickInfo += `<div class="mb-1">${item.thickIn || '-'}</div>`;
          if(item.thickOut) thickInfo += `<div>${item.thickOut}</div>`;
          glassInfo += `<div class="flex items-center mb-1"><span class="glass-tag tag-in shrink-0">내부</span> <span class="text-blue-600 font-bold">${item.glassIn || '-'}</span></div>`;
          if(item.glassOut && item.glassOut.trim() !== '') {
             glassInfo += `<div class="flex items-center"><span class="glass-tag tag-out shrink-0">외부</span> <span class="text-pink-700 font-bold">${item.glassOut}</span></div>`;
          }
        }

        let hideClass = isEtc ? 'hidden' : 'flex';

        let rowHtml = `
          <div class="p-5 hover:bg-gray-50 transition duration-150 group">
            <div class="flex justify-between items-start mb-3">
              <div class="flex items-center">
                <span class="bg-blue-100 text-blue-800 text-[10px] font-bold px-2 py-0.5 rounded mr-2 group-hover:bg-blue-600 group-hover:text-white transition">#${idx+1}</span>
                <h4 class="font-bold text-lg text-gray-800">${item.loc || '위치정보 없음'}</h4>
              </div>
              <span class="font-bold text-gray-900 text-lg">${fmt(item.price)}</span>
            </div>
            
            <div class="bg-white rounded-lg p-3 border border-gray-100 grid grid-cols-1 md:grid-cols-2 gap-x-4 gap-y-1">
               <div class="spec-row">
                 <span class="spec-label">제품명</span>
                 <span class="spec-value">${item.prod || '-'}</span>
               </div>
               <div class="spec-row ${hideClass}">
                 <span class="spec-label">규격</span>
                 <span class="spec-value">${item.size || '-'}</span>
               </div>
               <div class="spec-row">
                 <span class="spec-label">${displaySpecLabel}</span>
                 <span class="spec-value">${displaySpecValue}</span>
               </div>
               <div class="hidden md:block spec-row"></div>
               <div class="spec-row items-center ${hideClass}">
                 <span class="spec-label">두께</span>
                 <div class="spec-value text-xs font-medium text-gray-700">${thickInfo}</div>
               </div>
               <div class="spec-row items-center ${hideClass}">
                 <span class="spec-label">유리종류</span>
                 <div class="spec-value text-xs">${glassInfo}</div>
               </div>
            </div>
          </div>
        `;
        container.innerHTML += rowHtml;
      });
      
      if(items.length === 0) {
        container.innerHTML = `<div class="p-8 text-center text-gray-400 text-sm">상세 내역이 없습니다. PDF 견적서를 확인해주세요.</div>`;
      }
    }

    // --- [수정됨] 배너 렌더링 함수: 반응형 적용 (w-full h-auto) ---
    function renderBanners() {
      const container = document.getElementById('bannerContainer');
      container.innerHTML = "";

      if(g_banners.length === 0) {
        return;
      }

      g_banners.forEach(b => {
        // [수정됨] 고정 높이 제거, 가로 100%에 맞춰 높이 자동 조절
        let html = `
          <a href="${b.link}" target="_blank" class="block rounded-xl overflow-hidden shadow-sm hover:opacity-95 transition relative">
            <img src="${b.img}" class="w-full h-auto object-contain" alt="Banner">
          </a>
        `;
        container.innerHTML += html;
      });
    }
